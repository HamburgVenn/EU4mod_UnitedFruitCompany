UFC_staff_estate_tropical_nation = {
	icon = privilege_grant_new_world_charter
	loyalty = 0.05
	influence = 0.05
	is_valid = {
		OR = {
			tag = UFC
			has_estate_privilege = UFC_staff_estate_tropical_nation
		}
	}
	can_select = {
		capital_scope = {
			has_climate = tropical
		}
	}
	on_granted = {
		custom_tooltip = estate_burghers_tropical_effect_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				has_climate = tropical
			}
			add_province_modifier = {
				name = estate_burghers_tropical_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_burghers_tropical_mod
			}
			remove_province_modifier = estate_burghers_tropical_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_burghers_tropical_revoke_effect_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_burghers_tropical_mod
			}
			remove_province_modifier = estate_burghers_tropical_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_burghers_tropical_revoke_effect_tooltip
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_burghers_tropical_mod
			}
			remove_province_modifier = estate_burghers_tropical_mod
		}
	}
	ai_will_do = {
		factor = 1000
	}
}
